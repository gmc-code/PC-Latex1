\section*{Examples using mhchem}

\begin{enumerate}
\item Covalent compounds: \ce{H2O}
\item Ionic compounds: \ce{NaCl}
\item Hydrates: \ce{CuSO4.5H2O}
\item Stoichiometry: \ce{2H2 + O2 -> 2H2O}
\item Acids and bases: \ce{HCl + NaOH -> NaCl + H2O}
\item Chemical equation: \ce{H2O -> H+ + OH-}
\item Simple Isotopes: \ce{^{14}C}
\item Isotopes: \ce{^{227}_{90}Th+}
\item Reaction arrows: \ce{A ->[H2O] B}
\item States of matter: \ce{H2O(l) -> H2O(g)}
\item States of aggregation: \ce{H2(g) + O2(g) -> H2O(l)}
\item Charges: \ce{SO4^2-}
\item Charges: \ce{SO4^2- + Ba^2+ -> BaSO4 v}
\item Oxidation states: \ce{Fe^{3+} + Cr2O7^{2-} -> Fe^{2+} + Cr^{3+}}
\item Equilibrium arrows: \ce{CO2 + H2O <=> H2CO3}
\item Equilibrium arrows: \ce{CO2 + H2O <<=> H+ + HCO3-}
\item Precipitation: \ce{AgCl(s) <=> Ag+ + Cl-}
\item Acids and bases: \ce{HCl + NaOH -> NaCl + H2O}
\item Complex ions: \ce{[Cu(NH3)4(H2O)2]^{2+}}
\item Hydrates: \ce{CuSO4.5H2O}
\item Organic compounds: \ce{CH3CH2OH}
\item Polymers: \ce{(C2H4)_n}
\item Biochemical compounds: \ce{ATP + H2O -> ADP + Pi + H+}
\item Nuclear reactions: \ce{^{14}_6C -> ^{14}_7N + e- + $\bar{\nu}_e$}
\item Redox reactions: \ce{Zn + Cu^{2+} -> Zn^{2+} + Cu}
\item Ionic equations: \ce{Ag+(aq) + Cl-(aq) -> AgCl(s)}
\item Half-reactions:\\ 
Oxidation half-reaction: \ce{Zn -> Zn^{2+} + 2e-} \\
Reduction half-reaction: \ce{Cu^{2+} + 2e- -> Cu}
\item Reaction mechanisms:\\
Step 1: \ce{NO2 + NO3 -> NO + NO2 + O2} \\
Step 2: \ce{NO + O3 -> NO2 + O2} \\
Overall reaction: \ce{NO3 + O3 -> NO2 + O2 + O2}
\end {enumerate}

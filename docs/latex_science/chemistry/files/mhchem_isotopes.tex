\section*{Examples of isotope equations involving alpha decay, beta decay, and gamma radiation}

\begin{enumerate}
  \item Alpha decay: \ce{^{238}_{92}U -> ^{234}_{90}Th + ^{4}_{2}He}
  \item Beta decay: \ce{^{14}_{6}C -> ^{14}_{7}N + e- + $\bar{\nu}_e$}
  \item Gamma radiation: \ce{^{60}_{27}Co -> ^{60}_{28}Ni + e- + $\bar{\nu}_e$ + $\gamma$}
  \item Alpha decay: \ce{^{226}_{88}Ra -> ^{222}_{86}Rn + ^{4}_{2}He}
  \item Beta decay: \ce{^{40}_{19}K -> ^{40}_{20}Ca + e- + $\bar{\nu}_e$}
  \item Gamma radiation: \ce{^{99m}_{43}Tc -> ^{99}_{43}Tc + $\gamma$}
  \item Alpha decay: \ce{^{210}_{84}Po -> ^{206}_{82}Pb + ^{4}_{2}He}
  \item Beta decay: \ce{^{3}_{1}H -> ^{3}_{2}He + e- + $\bar{\nu}_e$}
  \item Gamma radiation: \ce{^{137m}_{55}Cs -> ^{137}_{55}Cs + $\gamma$}
  \item Alpha decay: \ce{^{235}_{92}U -> ^{231}_{90}Th + ^{4}_{2}He}
\end {enumerate}

% Define new variables to store the values of the angles
\pgfmathsetmacro{\angleAValue}{90}
\pgfmathsetmacro{\angleBValue}{int(rnd*40+25)}
\pgfmathsetmacro{\angleCValue}{int(180-\angleAValue-\angleBValue)}
\pgfmathsetmacro{\angleBCValue}{int(\angleBValue+\angleCValue)}
\pgfmathsetmacro{\sideCValue}{rnd*1+3}
\pgfmathsetmacro{\rotationAngleValue}{rnd*90}

\begin{document}
\section*{Answer}
    \begin{minipage}{0.55\textwidth}
        \begin{tikzpicture}[scale=1.5, baseline=(current bounding box.south)]
            \pgfmathsetmacro{\angleA}{\angleAValue}
            \pgfmathsetmacro{\angleB}{\angleBValue}
            \pgfmathsetmacro{\sideC}{\sideCValue}
            \pgfmathsetmacro{\rotationAngle}{\rotationAngleValue}
            \pgfmathsetmacro{\angleC}{\angleCValue}
          
            \begin{scope}[rotate=\rotationAngle]
              \coordinate (A) at (0,0);
              \coordinate (B) at (\sideC,0);
              \coordinate (C) at (intersection cs: first line={(A)--($(A)+(\angleA:4cm)$)}, second line={(B)--($(B)+(180-\angleB:4cm)$)});
              \draw (A) -- (B) -- (C) -- cycle;
              
              % Mark angles with arcs
              \draw ($(A)!0.3cm!(B)$) arc [start angle=0, end angle=\angleA, radius=0.3cm];
              \draw ($(B)!0.3cm!(C)$) arc [start angle=180-\angleB, end angle=180, radius=0.3cm];
              %   \draw ($(C)!0.3cm!(A)$) arc [start angle=180+\angleA, end angle=360-\angleB, radius=0.3cm];
              % Add a right angle symbol at vertex A
              \draw ($(A)!0.3cm!(B)$) -- ++(90:0.3cm) -- ++(180:0.3cm);

              % Label angles
              \node at ($(A)!-0.15cm!(B)$) {A};
              \node at ($(B)!-0.15cm!(C)$) {B};
              \node at ($(C)!-0.15cm!(A)$) {C};
              
              % Mark angles in degrees
              \coordinate (midBC) at ($(B)!0.5!(C)$);
              \node at ($(A)!0.55cm!(midBC)$) {$\theta ^\circ$};
          
              \coordinate (midAC) at ($(A)!0.5!(C)$);
              \node at ($(B)!0.55cm!(midAC)$) {\angleBValue$^\circ$};
          
              \coordinate (midAB) at ($(A)!0.5!(B)$);
              \node at ($(C)!0.55cm!(midAB)$) {\angleCValue$^\circ$};
           
            \end{scope}
          \end{tikzpicture}
\end{minipage}%
\hfill
\begin{minipage}{0.4\textwidth}
    \begin{align*}
        \angle \text{A} &= 180^\circ - (\angle \text{B} + \angle \text{C}) \\
        &= 180^\circ - (\angleBValue^\circ + \angleCValue^\circ) \\
        &= 180^\circ - \angleBCValue^\circ \\
        &= \angleAValue^\circ
    \end{align*}
\end{minipage}

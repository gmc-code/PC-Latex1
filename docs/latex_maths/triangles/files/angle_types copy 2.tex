\documentclass{article}
\usepackage{pgf}
\usepackage{pgffor} % for the \foreach command
\usepackage[first=1, last=6]{lcg} % for the \rand command

\pgfmathsetseed{\number\pdfrandomseed} % to ensure that it is randomized
\newcommand{\mylist}{1,2,3,4,5,6} % define the original list
\pgfmathdeclarerandomlist{mylist}{\mylist} % declare the list as a random list
\begin{document}
\foreach \i in {1,...,6} { % loop over the list size
\pgfmathrandomitemwithoutreplacement{\x}{mylist} % get a random item from the list without repeating it
\x % print the item
}
\end{document}

% Acute angle
\newcommand{\acuteangle}[1]{
  \draw (1,0) coordinate (B) -- (0,0) coordinate (A) -- (1,1) coordinate (C);
  \pic [draw, "$\theta$", angle eccentricity=1.5] {angle = B--A--C};
  \ifthenelse{\equal{#1}{true}}
  {\node at (0.5,-0.25) {Acute angle};}
  {\node at (0.5,-0.25) {\dotuline{~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~}};}
}

% Right angle
\newcommand{\rightangle}[1]{
  \draw (0,1) coordinate (B) -- (0,0) coordinate (A) -- (1,0) coordinate (C);
  \pic [draw, "$\theta$", angle eccentricity=-1.5] {right angle = B--A--C};
  \ifthenelse{\equal{#1}{true}}
  {\node at (0.5,-0.25) {Right angle};}
  {\node at (0.5,-0.25) {\dotuline{~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~}};}
}

% Revolution
\newcommand{\revolution}[1]{
  \draw (0,0) coordinate (A) -- (1,0) coordinate (B);
  \draw (A) +(0:0.10) arc [start angle=0, end angle=360, radius=0.10];
  \node at (0.15,0.15) {$\theta$};
  \ifthenelse{\equal{#1}{true}}
  {\node at (0.5,-0.25) {Revolution};}
  {\node at (0.5,-0.25) {\dotuline{~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~}};}
}
